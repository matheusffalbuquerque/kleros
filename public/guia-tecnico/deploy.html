#!/bin/bash
# setup-server.sh - Configuração completa do servidor para Kleros

set -e

echo "🚀 Iniciando configuração do servidor Kleros..."

# Atualizar sistema
apt update && apt upgrade -y

# Instalar dependências básicas
apt install -y software-properties-common apt-transport-https ca-certificates \
    curl gnupg lsb-release unzip git supervisor

# Adicionar repositórios
add-apt-repository ppa:ondrej/php -y
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

# Instalar PHP 8.2
apt update
apt install -y php8.2 php8.2-fpm php8.2-mysql php8.2-redis php8.2-xml \
    php8.2-gd php8.2-mbstring php8.2-curl php8.2-zip php8.2-bcmath \
    php8.2-intl php8.2-soap php8.2-cli

# Instalar Composer
curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer

# Instalar Node.js
curl -fsSL https://deb.nodesource.com/setup_20.x | bash -
apt install -y nodejs

# Instalar Nginx
apt install -y nginx

# Instalar MySQL
apt install -y mysql-server

# Configurar MySQL
mysql_secure_installation

# Instalar Redis
apt install -y redis-server

# Configurar Redis
systemctl enable redis-server
systemctl start redis-server

# Configurar Firewall
ufw allow ssh
ufw allow 'Nginx Full'
ufw --force enable

# Criar usuário para aplicação
adduser --disabled-password --gecos "" kleros
usermod -aG www-data kleros

# Configurar diretórios
mkdir -p /var/www/kleros
chown kleros:www-data /var/www/kleros

echo "✅ Configuração básica do servidor concluída!"

    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name ~^(?.+)\.kleros\.app$;
    
    root /var/www/kleros/public;
    index index.php;
    
    # SSL Configuration
    ssl_certificate /etc/letsencrypt/live/kleros.app/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/kleros.app/privkey.pem;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    
    # Security Headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline';" always;
    
    # Gzip Compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/xml+rss application/json;
    
    # Static Files
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        try_files $uri =404;
    }
    
    # Laravel Application
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }
    
    location = /favicon.ico { access_log off; log_not_found off; }
    location = /robots.txt  { access_log off; log_not_found off; }
    
    error_page 404 /index.php;
    
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
        fastcgi_hide_header X-Powered-By;
        
        # Passar subdomínio para aplicação
        fastcgi_param HTTP_HOST $subdomain.kleros.app;
    }
    
    location ~ /\.(?!well-known).* {
        deny all;
    }
    
    # Rate Limiting
    location /api/ {
        limit_req zone=api burst=20 nodelay;
        try_files $uri $uri/ /index.php?$query_string;
    }
    
    # Admin Panel Protection
    location /admin {
        # Permitir apenas IPs específicos (opcional)
        # allow 203.0.113.0/24;
        # deny all;
        
        try_files $uri $uri/ /index.php?$query_string;
    }
}

📦 Script de Deploy

Automatização do processo de deploy

#!/bin/bash
# deploy.sh - Script de deploy automatizado para Kleros

set -e

# Configurações
REPO_URL="https://github.com/seu-usuario/kleros.git"
DEPLOY_PATH="/var/www/kleros"
PHP_VERSION="8.2"
BACKUP_PATH="/var/backups/kleros"

# Cores para output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

echo_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

echo_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# Verificar se está executando como root
if [[ $EUID -eq 0 ]]; then
   echo_error "Este script não deve ser executado como root"
   exit 1
fi

echo_info "🚀 Iniciando deploy do Kleros..."

# Verificar se o diretório existe
if [ ! -d "$DEPLOY_PATH" ]; then
    echo_error "Diretório de deploy não encontrado: $DEPLOY_PATH"
    exit 1
fi

cd $DEPLOY_PATH

# Colocar aplicação em modo de manutenção
echo_info "📝 Colocando aplicação em modo de manutenção..."
php artisan down --render="errors.503" --secret="deploy-secret-123"

# Fazer backup do banco de dados
echo_info "💾 Fazendo backup do banco de dados..."
mkdir -p $BACKUP_PATH
TIMESTAMP=$(date +"%Y%m%d_%H%M%S")
php artisan backup:run --only-db --filename="backup_${TIMESTAMP}.sql"

# Backup dos arquivos atuais
echo_info "📁 Fazendo backup dos arquivos..."
tar -czf "$BACKUP_PATH/files_backup_${TIMESTAMP}.tar.gz" \
    --exclude='node_modules' \
    --exclude='vendor' \
    --exclude='.git' \
    -C /var/www/ kleros/

# Fazer pull das alterações
echo_info "📥 Atualizando código..."
git fetch origin
git reset --hard origin/main

# Verificar se composer.json foi modificado
if git diff --name-only HEAD~1 HEAD | grep -q "composer.json\|composer.lock"; then
    echo_info "📦 Instalando dependências PHP..."
    composer install --no-dev --optimize-autoloader --no-interaction
fi

# Verificar se package.json foi modificado
if git diff --name-only HEAD~1 HEAD | grep -q "package.json\|package-lock.json"; then
    echo_info "📦 Instalando dependências Node.js..."
    npm ci --production
fi

# Build dos assets
echo_info "🔨 Compilando assets..."
npm run build

# Limpar caches
echo_info "🧹 Limpando caches..."
php artisan config:clear
php artisan route:clear
php artisan view:clear
php artisan cache:clear

# Otimizar para produção
echo_info "⚡ Otimizando para produção..."
php artisan config:cache
php artisan route:cache
php artisan view:cache
php artisan event:cache

# Executar migrations
echo_info "🗃️ Executando migrations..."
php artisan migrate --force

# Ajustar permissões
echo_info "🔐 Ajustando permissões..."
sudo chown -R kleros:www-data $DEPLOY_PATH
sudo chmod -R 755 $DEPLOY_PATH
sudo chmod -R 775 $DEPLOY_PATH/storage $DEPLOY_PATH/bootstrap/cache

# Recarregar PHP-FPM
echo_info "🔄 Recarregando PHP-FPM..."
sudo systemctl reload php${PHP_VERSION}-fpm

# Reiniciar workers de queue
echo_info "👷 Reiniciando workers..."
sudo supervisorctl restart kleros-worker:*

# Aguardar alguns segundos
sleep 5

# Verificar se a aplicação está funcionando
echo_info "🔍 Verificando aplicação..."
if curl -f -s -I "https://admin.kleros.app/health" > /dev/null; then
    echo_info "✅ Aplicação está respondendo!"
else
    echo_error "❌ Aplicação não está respondendo!"
    echo_warn "🔄 Revertendo deploy..."
    
    # Revert para commit anterior
    git reset --hard HEAD~1
    php artisan config:cache
    php artisan route:cache
    sudo supervisorctl restart kleros-worker:*
    
    echo_error "Deploy foi revertido devido a erro"
    exit 1
fi

# Retirar do modo de manutenção
echo_info "🟢 Retirando aplicação do modo de manutenção..."
php artisan up

# Limpar backups antigos (manter apenas os últimos 10)
echo_info "🧹 Limpando backups antigos..."
ls -t $BACKUP_PATH/backup_*.sql | tail -n +11 | xargs -r rm
ls -t $BACKUP_PATH/files_backup_*.tar.gz | tail -n +11 | xargs -r rm

echo_info "🎉 Deploy concluído com sucesso!"
echo_info "📊 Informações do deploy:"
echo "   📅 Data: $(date)"
echo "   🏷️  Commit: $(git rev-parse --short HEAD)"
echo "   👤 Autor: $(git log -1 --pretty=format:'%an')"
echo "   📝 Mensagem: $(git log -1 --pretty=format:'%s')"

# Notificar equipe (opcional)
# curl -X POST -H 'Content-type: application/json' \
#     --data '{"text":"🚀 Deploy do Kleros realizado com sucesso!"}' \
#     $SLACK_WEBHOOK_URL

Monitoramento

# Script de monitoramento - /usr/local/bin/monitor-kleros.sh
#!/bin/bash

LOG_FILE="/var/log/kleros-monitor.log"
DATE=$(date '+%Y-%m-%d %H:%M:%S')

log_message() {
    echo "[$DATE] $1" >> $LOG_FILE
}

# Verificar workers
WORKERS_RUNNING=$(supervisorctl status kleros-worker:* | grep RUNNING | wc -l)
WORKERS_EXPECTED=4

if [ $WORKERS_RUNNING -lt $WORKERS_EXPECTED ]; then
    log_message "ALERT: Apenas $WORKERS_RUNNING de $WORKERS_EXPECTED workers rodando"
    supervisorctl restart kleros-worker:*
fi

# Verificar uso de memória
MEMORY_USAGE=$(free | grep Mem | awk '{printf "%.0f", $3/$2 * 100.0}')
if [ $MEMORY_USAGE -gt 80 ]; then
    log_message "WARNING: Uso de memória alto: ${MEMORY_USAGE}%"
fi

# Verificar espaço em disco
DISK_USAGE=$(df / | tail -1 | awk '{print $5}' | sed 's/%//')
if [ $DISK_USAGE -gt 85 ]; then
    log_message "WARNING: Uso de disco alto: ${DISK_USAGE}%"
fi

# Verificar se aplicação está respondendo
if ! curl -f -s "https://admin.kleros.app/health" > /dev/null; then
    log_message "CRITICAL: Aplicação não está respondendo"
    # Reiniciar PHP-FPM
    systemctl restart php8.2-fpm
fi

# Verificar logs de erro
ERROR_COUNT=$(tail -100 /var/www/kleros/storage/logs/laravel.log | grep -c ERROR)
if [ $ERROR_COUNT -gt 10 ]; then
    log_message "WARNING: $ERROR_COUNT erros nos últimos logs"
fi

# Adicionar ao crontab
# */5 * * * * /usr/local/bin/monitor-kleros.sh

💾 Backup e Recuperação

Estratégia de backup automatizado

#!/bin/bash
# backup-kleros.sh - Script de backup automatizado

BACKUP_DIR="/var/backups/kleros"
S3_BUCKET="kleros-backups"
MYSQL_USER="backup_user"
MYSQL_PASS="senha_segura"
RETENTION_DAYS=30

TIMESTAMP=$(date +"%Y%m%d_%H%M%S")
BACKUP_NAME="kleros_backup_${TIMESTAMP}"

mkdir -p $BACKUP_DIR

echo "🗃️ Iniciando backup do Kleros..."

# Backup do banco de dados
echo "📊 Fazendo backup do banco de dados..."
mysqldump --user=$MYSQL_USER --password=$MYSQL_PASS \
    --single-transaction --routines --triggers \
    --all-databases > "$BACKUP_DIR/${BACKUP_NAME}.sql"

# Compactar backup do banco
gzip "$BACKUP_DIR/${BACKUP_NAME}.sql"

# Backup dos arquivos da aplicação
echo "📁 Fazendo backup dos arquivos..."
tar -czf "$BACKUP_DIR/${BACKUP_NAME}_files.tar.gz" \
    --exclude='node_modules' \
    --exclude='vendor' \
    --exclude='.git' \
    --exclude='storage/logs/*.log' \
    -C /var/www/ kleros/

# Backup dos uploads/storage
echo "📸 Fazendo backup dos uploads..."
tar -czf "$BACKUP_DIR/${BACKUP_NAME}_storage.tar.gz" \
    -C /var/www/kleros/storage/app/ public/

# Upload para S3 (se configurado)
if command -v aws &> /dev/null; then
    echo "☁️ Enviando backups para S3..."
    aws s3 cp "$BACKUP_DIR/${BACKUP_NAME}.sql.gz" "s3://$S3_BUCKET/database/"
    aws s3 cp "$BACKUP_DIR/${BACKUP_NAME}_files.tar.gz" "s3://$S3_BUCKET/files/"
    aws s3 cp "$BACKUP_DIR/${BACKUP_NAME}_storage.tar.gz" "s3://$S3_BUCKET/storage/"
fi

# Limpeza de backups antigos
echo "🧹 Removendo backups antigos..."
find $BACKUP_DIR -name "kleros_backup_*" -mtime +$RETENTION_DAYS -delete

# Verificar integridade dos backups
echo "✅ Verificando integridade..."
if gzip -t "$BACKUP_DIR/${BACKUP_NAME}.sql.gz" && \
   tar -tzf "$BACKUP_DIR/${BACKUP_NAME}_files.tar.gz" > /dev/null && \
   tar -tzf "$BACKUP_DIR/${BACKUP_NAME}_storage.tar.gz" > /dev/null; then
    echo "✅ Backup concluído com sucesso!"
else
    echo "❌ Erro na verificação do backup!"
    exit 1
fi

# Log do backup
echo "$(date): Backup $BACKUP_NAME concluído" >> /var/log/kleros-backup.log

# Configurar cron para backup diário
# 0 2 * * * /usr/local/bin/backup-kleros.sh

# Script de recuperação
cat > /usr/local/bin/restore-kleros.sh << 'EOF'
#!/bin/bash
# restore-kleros.sh - Script de recuperação

if [ $# -eq 0 ]; then
    echo "Uso: $0 "
    echo "Exemplo: $0 kleros_backup_20241125_020000"
    exit 1
fi

BACKUP_NAME=$1
BACKUP_DIR="/var/backups/kleros"

echo "🔄 Iniciando recuperação do backup: $BACKUP_NAME"

# Colocar em manutenção
cd /var/www/kleros
php artisan down

# Restaurar banco de dados
echo "📊 Restaurando banco de dados..."
gunzip -c "$BACKUP_DIR/${BACKUP_NAME}.sql.gz" | mysql

# Restaurar arquivos
echo "📁 Restaurando arquivos..."
cd /var/www/
tar -xzf "$BACKUP_DIR/${BACKUP_NAME}_files.tar.gz"

# Restaurar storage
echo "📸 Restaurando storage..."
cd /var/www/kleros/storage/app/
tar -xzf "$BACKUP_DIR/${BACKUP_NAME}_storage.tar.gz"

# Ajustar permissões
chown -R kleros:www-data /var/www/kleros
chmod -R 755 /var/www/kleros
chmod -R 775 /var/www/kleros/storage /var/www/kleros/bootstrap/cache

# Retirar de manutenção
cd /var/www/kleros
php artisan up

echo "✅ Recuperação concluída!"
EOF

chmod +x /usr/local/bin/restore-kleros.sh

// app/Console/Commands/QueueHeartbeat.php
class QueueHeartbeat extends Command
{
    protected $signature = 'queue:heartbeat';

    {

    }
}

Log Monitoring

# Script de análise de logs
#!/bin/bash
# log-analyzer.sh

LOG_FILE="/var/www/kleros/storage/logs/laravel.log"
ALERT_EMAIL="admin@kleros.app"
TEMP_FILE="/tmp/log_analysis.txt"

# Verificar erros nas últimas 24h
grep "$(date -d '1 day ago' '+%Y-%m-%d')" $LOG_FILE | grep "ERROR\|CRITICAL" > $TEMP_FILE

ERROR_COUNT=$(wc -l < $TEMP_FILE)

if [ $ERROR_COUNT -gt 50 ]; then
    echo "ALERTA: $ERROR_COUNT erros encontrados nas últimas 24h" | \
    mail -s "Kleros - Muitos erros detectados" $ALERT_EMAIL
fi

# Top 10 erros mais frequentes
echo "=== TOP 10 ERROS ===" >> $TEMP_FILE
grep "ERROR" $LOG_FILE | \
    sed 's/\[.*\]//' | \
    sort | uniq -c | sort -nr | head -10 >> $TEMP_FILE

# Configurar logrotate para Laravel
cat > /etc/logrotate.d/laravel << 'EOF'
/var/www/kleros/storage/logs/*.log {
    daily
    missingok
    rotate 30
    compress
    delaycompress
    notifempty
    create 644 kleros www-data
    postrotate
        # Limpar caches de log do Laravel
        php /var/www/kleros/artisan cache:clear > /dev/null 2>&1 || true
    endscript
}
EOF

# Configurar rsyslog para centralizar logs
cat >> /etc/rsyslog.conf << 'EOF'
# Kleros logs
$template KlerosFormat,"%TIMESTAMP% %HOSTNAME% %syslogtag% %msg%\n"
:programname, isequal, "kleros" /var/log/kleros/app.log;KlerosFormat
& stop
EOF

systemctl restart rsyslog
