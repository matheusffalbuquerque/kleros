class User extends Authenticatable
{
    use HasRoles, Notifiable;
    
    protected $fillable = [
        'name', 'email', 'password',
        'congregacao_id', 'denominacao_id'
    ];
    
    // Relacionamentos
    public function congregacao() {
        return $this->belongsTo(Congregacao::class);
    }
    
    public function membro() {
        return $this->belongsTo(Membro::class);
    }
    
    // Roles: kleros, admin, gestor
}

👥 Membro Model

class Membro extends Model
{
    protected $table = 'membros';

        'congregacao_id', 'primeiro_nome',
        'sobrenome', 'email', 'telefone',
        'data_nascimento', 'sexo', 'ativo'

        'data_nascimento' => 'date',

    }

🔍 Query Scopes

Reutilização de consultas comuns

Local Scopes

class Membro extends Model
{
    // Scope para membros ativos
    public function scopeAtivos($query) {
        return $query->where('ativo', true);
    }
    
    // Scope por sexo
    public function scopePorSexo($query, $sexo) {
        return $query->where('sexo', $sexo);
    }
    
    // Scope por faixa etária
    public function scopePorIdade($query, $min, $max) {
        return $query->whereBetween('data_nascimento', [
            now()->subYears($max),
            now()->subYears($min)
        ]);
    }
}

// Uso dos scopes
$membros = Membro::ativos()
    ->porSexo('M')
    ->porIdade(18, 65)
    ->get();

Global Scopes

// Scope global para filtrar por congregação
class CongregacaoScope implements Scope
{
    public function apply(Builder $builder, Model $model)
    {
        if (app()->bound('congregacao')) {
            $congregacao = app('congregacao');
            $builder->where('congregacao_id', $congregacao->id);
        }
    }
}

// Aplicar no model
class Membro extends Model
{
    protected static function booted()
    {
        static::addGlobalScope(new CongregacaoScope);
    }
}

🔄 Accessors & Mutators

Transformação de dados nos Models

Accessors (Get)

class Membro extends Model
{
    // Nome completo
    public function getNomeCompletoAttribute()
    {
        return $this->primeiro_nome . ' ' . $this->sobrenome;
    }
    
    // Idade calculada
    public function getIdadeAttribute()
    {
        return $this->data_nascimento
            ? $this->data_nascimento->age
            : null;
    }
    
    // Avatar padrão
    public function getAvatarUrlAttribute()
    {
        return $this->avatar 
            ? Storage::url($this->avatar)
            : 'https://ui-avatars.com/api/?name=' . urlencode($this->nome_completo);
    }
}

// Uso
echo $membro->nome_completo;
echo $membro->idade;
echo $membro->avatar_url;

Mutators (Set)

class User extends Model
{
    // Hash automático da senha
    public function setPasswordAttribute($value)
    {
        $this->attributes['password'] = Hash::make($value);
    }
    
    // Normalização de email
    public function setEmailAttribute($value)
    {
        $this->attributes['email'] = strtolower(trim($value));
    }
}

class Membro extends Model
{
    // Formatação de telefone
    public function setTelefoneAttribute($value)
    {
        $this->attributes['telefone'] = preg_replace('/\D/', '', $value);
    }
}
