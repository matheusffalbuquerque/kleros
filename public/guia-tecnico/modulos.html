🧩 Sistema Modular

Arquitetura extensível que permite adicionar funcionalidades específicas através de módulos independentes.

💡 Conceito e Arquitetura

Como funciona o sistema modular do Kleros

O que são Módulos?

Módulos são extensões funcionais independentes que podem ser ativadas ou desativadas por congregação.

Características:
  • 🔧 Auto-contidos (MVC próprio)
  • 🎛️ Ativação por congregação
  • 🔌 Service Providers próprios
  • 📊 Migrations independentes

Validação e Carregamento

// ExtensoesController.php
class ExtensoesController extends Controller
{
    public function index()
    {
        $modulesPath = base_path('modules');
        $congregacaoId = app('congregacao')->id;
        
        // Estados do banco
        $databaseStates = Extensao::where('congregacao_id', $congregacaoId)
            ->get()
            ->keyBy('module');
        
        // Scan de módulos no filesystem
        $modules = collect(File::directories($modulesPath))
            ->map(function ($directory) use ($databaseStates) {
                $manifest = $this->loadManifest($directory);
                $moduleKey = strtolower(basename($directory));
                
                return [
                    'name' => $manifest['name'] ?? 'Módulo',
                    'slug' => $moduleKey,
                    'description' => $manifest['description'] ?? '',
                    'version' => $manifest['version'] ?? '1.0.0',
                    'enabled' => $databaseStates->get($moduleKey)?->enabled ?? false,
                    'config' => $manifest['config'] ?? []
                ];
            });
            
        return view('extensoes.index', compact('modules'));
    }
    
    private function loadManifest($directory)
    {
        $manifestPath = $directory . '/module.json';
        
        if (!File::exists($manifestPath)) {
            return [];
        }
        
        return json_decode(File::get($manifestPath), true) ?? [];
    }
}

    /**
     * Criar nova célula
     */
    public function store(Request $request)
    {
        $validated = $request->validate([
            'nome' => 'required|string|max:255',
            'endereco' => 'nullable|string',
            'dia_semana' => 'required|integer|between:0,6',
            'horario' => 'required|date_format:H:i',
            'lider_id' => 'required|exists:membros,id',
            'ativa' => 'boolean'
        ]);
        
        $validated['congregacao_id'] = app('congregacao')->id;
        
        $celula = Celula::create($validated);
        
        // Adicionar líder automaticamente
        $celula->membros()->attach($validated['lider_id'], [
            'funcao' => 'lider',
            'data_ingresso' => now()
        ]);
        
        return redirect()->route('celulas.painel')
            ->with('success', 'Célula criada com sucesso!');
    }
    
    /**
     * Busca AJAX
     */
    public function search(Request $request)
    {
        $query = $request->get('q', '');
        $congregacao = app('congregacao');
        
        $celulas = $congregacao->celulas()
            ->where('nome', 'like', "%{$query}%")
            ->where('ativa', true)
            ->select('id', 'nome', 'endereco')
            ->limit(10)
            ->get();
            
        return response()->json($celulas);
    }
}

🎛️ Sistema de Ativação

Como congregações ativam/desativam módulos

Model Extensao

// Migration: create_extensoes_table
Schema::create('extensoes', function (Blueprint $table) {
    $table->id();
    $table->foreignId('congregacao_id')
          ->constrained('congregacoes')
          ->onDelete('cascade');
    $table->string('module');
    $table->boolean('enabled')->default(false);
    $table->json('config')->nullable();
    $table->timestamps();
    
    $table->unique(['congregacao_id', 'module']);
    $table->index(['congregacao_id', 'enabled']);
});

// Model
class Extensao extends Model
{
    protected $fillable = [
        'congregacao_id', 'module', 'enabled', 'config'
    ];
    
    protected $casts = [
        'enabled' => 'boolean',
        'config' => 'array'
    ];
    
    public function congregacao()
    {
        return $this->belongsTo(Congregacao::class);
    }
    
    // Scope para módulos ativos
    public function scopeEnabled($query)
    {
        return $query->where('enabled', true);
    }
}

Controller de Extensões

class ExtensoesController extends Controller
{
    /**
     * Ativar/desativar módulo
     */
    public function update(Request $request, $module)
    {
        $request->validate([
            'enabled' => 'required|boolean',
            'config' => 'nullable|array'
        ]);
        
        $congregacao = app('congregacao');
        
        // Atualizar ou criar registro
        Extensao::updateOrCreate(
            [
                'congregacao_id' => $congregacao->id,
                'module' => $module
            ],
            [
                'enabled' => $request->enabled,
                'config' => $request->config ?? []
            ]
        );
        
        // Log da atividade
        Log::info('Módulo ' . ($request->enabled ? 'ativado' : 'desativado'), [
            'module' => $module,
            'congregacao' => $congregacao->identificacao,
            'user' => auth()->user()->name
        ]);
        
        return redirect()->route('extensoes.painel')
            ->with('success', 'Módulo atualizado com sucesso!');
    }
    
    /**
     * Verificar se módulo está ativo
     */
    public static function isModuleEnabled($module)
    {
        if (!app()->bound('congregacao')) {
            return false;
        }
        
        $congregacao = app('congregacao');
        
        return Extensao::where('congregacao_id', $congregacao->id)
            ->where('module', $module)
            ->where('enabled', true)
            ->exists();
    }
}

// Middleware para proteger rotas de módulos
class ModuleEnabled
{
    public function handle($request, Closure $next, $module)
    {
        if (!ExtensoesController::isModuleEnabled($module)) {
            abort(404, 'Módulo não está ativo para esta congregação.');
        }
        
        return $next($request);
    }
}

🛠️ Desenvolvendo Módulos

Guia para criar novos módulos

# 1. CRIAR ESTRUTURA DO MÓDULO
mkdir -p modules/MeuModulo/{Http/Controllers,routes,resources/views,database/migrations,Providers}

# 2. CRIAR MANIFESTO
cat > modules/MeuModulo/module.json << 'EOF'
{
    "name": "Meu Módulo",
    "slug": "meumodulo",
    "description": "Descrição do módulo",
    "version": "1.0.0",
    "author": "Seu Nome",
    "providers": [
        "Modules\\MeuModulo\\Providers\\MeuModuloServiceProvider"
    ]
}
EOF

# 3. CRIAR SERVICE PROVIDER
php artisan make:provider MeuModuloServiceProvider

# 4. REGISTRAR NO CONFIG/APP.PHP
'providers' => [
    // ...
    Modules\MeuModulo\Providers\MeuModuloServiceProvider::class,
],

# 5. CRIAR MIGRATION
php artisan make:migration create_meumodulo_table

# 6. EXECUTAR MIGRAÇÃO
php artisan migrate

# 7. TESTAR MÓDULO
- Acessar painel de extensões
- Ativar o módulo
- Testar funcionalidades
