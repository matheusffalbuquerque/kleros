🔐 Autenticação e Autorização

Sistema completo de autenticação, permissões baseadas em papéis e controle de acesso granular.

🎯 Visão Geral do Sistema

Como funciona a segurança no Kleros

Componentes Principais

  • 🔐 Autenticação: Laravel Breeze + Sanctum
  • 🎭 Autorização: Spatie Laravel Permission
  • 🛡️ Middleware: Controle de acesso por rota
  • 🏢 Multi-tenancy: Isolamento por congregação
  • 🔑 Recuperação: Códigos temporários via email

Fluxo de Autenticação

1. Usuário acessa {congregacao}.kleros.app
2. Middleware 'dominio' identifica congregação
3. Se não autenticado → redirecionado para /login
4. Formulário de login validado
5. Credenciais verificadas no banco
6. Sessão criada com dados do usuário
7. Permissões carregadas via Spatie
8. Redirecionamento para dashboard

Middleware Stack:
- web (sessões, CSRF)
- dominio (multi-tenancy)
- auth (autenticação)
- verified (email verificado)
- role:gestor (permissão específica)

class AuthServiceProvider extends ServiceProvider
{
    public function boot()
    {
        $this->registerPolicies();

        // Gate para criação de membros com lógica customizada
        Gate::define('create-membro', function (User $user) {
            // Além da permissão, verificar regras de negócio
            if (!$user->can('membros.create')) {
                return false;
            }

            // Limite de membros por congregação (exemplo)
            $congregacao = $user->congregacao();
            if ($congregacao && $congregacao->membros()->count() >= $congregacao->limite_membros) {
                return false;
            }

            return true;
        });

        // Gate para edição própria
        Gate::define('edit-own-profile', function (User $user, Membro $membro) {
            return $user->membro_id === $membro->id;
        });
    }

        // Rate limiting
        $throttleKey = $request->ip() . '|' . $request->email;
        if (RateLimiter::tooManyAttempts($throttleKey, 5)) {
            $seconds = RateLimiter::availableIn($throttleKey);
            return back()->withErrors([
                'email' => "Muitas tentativas. Tente novamente em {$seconds} segundos."
            ]);
        }

        $credentials = $request->only('email', 'password');

        // Verificar se usuário existe e está ativo
        $user = User::where('email', $credentials['email'])->first();
        
        if (!$user || !$user->is_active) {
            RateLimiter::hit($throttleKey);
            return back()->withErrors([
                'email' => 'Credenciais inválidas ou usuário inativo.'
            ]);
        }

        // Verificar se pode acessar esta congregação
        $congregacao = app('congregacao');
        if (!$user->podeAcessarCongregacao($congregacao)) {
            RateLimiter::hit($throttleKey);
            return back()->withErrors([
                'email' => 'Usuário não tem acesso a esta congregação.'
            ]);
        }

        // Tentar autenticar
        if (auth()->attempt($credentials, $request->boolean('remember'))) {
            $request->session()->regenerate();
            
            // Atualizar último login
            $user->update(['last_login_at' => now()]);
            
            RateLimiter::clear($throttleKey);
            
            return redirect()->intended('/dashboard');
        }

        RateLimiter::hit($throttleKey);
        
        return back()->withErrors([
            'email' => 'Credenciais inválidas.'
        ]);
    }

    /**

    public function logout(Request $request)
    {
        auth()->logout();
        $request->session()->invalidate();
        $request->session()->regenerateToken();
        
        return redirect('/login');
    }

    /**
     * Solicitar recuperação de senha
     */
    public function solicitarRecuperacao(Request $request)
    {
        $request->validate([
            'email' => 'required|email|exists:users,email'
        ]);

        $user = User::where('email', $request->email)->first();
        $congregacao = app('congregacao');

        // Verificar se usuário pertence a esta congregação
        if (!$user->podeAcessarCongregacao($congregacao)) {
            return back()->withErrors([
                'email' => 'Email não encontrado nesta congregação.'
            ]);
        }

        // Invalidar códigos anteriores
        $user->codigosRecuperacao()
             ->where('tipo', CodigoRecuperacao::TIPO_RESET_SENHA)
             ->validos()
             ->update(['usado_em' => now()]);

        // Criar novo código
        $codigo = CodigoRecuperacao::create([
            'user_id' => $user->id,
            'codigo' => CodigoRecuperacao::gerarCodigo(),
            'tipo' => CodigoRecuperacao::TIPO_RESET_SENHA,
            'expires_at' => now()->addMinutes(30)
        ]);

        // Enviar email
        Mail::to($user->email)->send(new CodigoRecuperacaoMail($codigo));

        return back()->with('status', 'Código de recuperação enviado por email.');
    }

    /**
     * Redefinir senha com código
     */
    public function redefinirSenha(Request $request)
    {
        $request->validate([
            'email' => 'required|email',
            'codigo' => 'required|string|size:6',
            'password' => 'required|string|min:8|confirmed'
        ]);

        $user = User::where('email', $request->email)->first();
        
        if (!$user) {
            return back()->withErrors(['email' => 'Email não encontrado.']);
        }

        $codigoModel = $user->codigosRecuperacao()
                           ->where('codigo', $request->codigo)
                           ->where('tipo', CodigoRecuperacao::TIPO_RESET_SENHA)
                           ->validos()
                           ->first();

        if (!$codigoModel) {
            return back()->withErrors(['codigo' => 'Código inválido ou expirado.']);
        }

        // Atualizar senha
        $user->update([
            'password' => Hash::make($request->password)
        ]);

        // Marcar código como usado
        $codigoModel->marcarComoUsado();

        return redirect()->route('login')->with('status', 'Senha redefinida com sucesso!');
    }
}

// RoleMiddleware
class RoleMiddleware
{
    public function handle(Request $request, Closure $next, $role, $guard = null)
    {
        $authGuard = app('auth')->guard($guard);

        if ($authGuard->guest()) {
            throw UnauthorizedException::notLoggedIn();
        }

        $roles = is_array($role)
            ? $role
            : explode('|', $role);

        if (!$authGuard->user()->hasAnyRole($roles)) {
            throw UnauthorizedException::forRoles($roles);
        }

        return $next($request);
    }
}

// Cliente JavaScript
class KlerosAPI {
    constructor(baseURL) {
        this.baseURL = baseURL;
        this.token = localStorage.getItem('auth_token');
    }

    async login(email, password, deviceName = 'Web Browser') {
        const response = await fetch(`${this.baseURL}/api/login`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Accept': 'application/json'
            },
            body: JSON.stringify({
                email,
                password,
                device_name: deviceName
            })
        });

        if (response.ok) {
            const data = await response.json();
            this.token = data.access_token;
            localStorage.setItem('auth_token', this.token);
            return data;
        }

        throw new Error('Login failed');
    }

    async logout() {
        await this.makeRequest('/api/logout', 'POST');
        this.token = null;
        localStorage.removeItem('auth_token');
    }

    async makeRequest(endpoint, method = 'GET', body = null) {
        const headers = {
            'Accept': 'application/json',
            'Authorization': `Bearer ${this.token}`
        };

        if (body) {
            headers['Content-Type'] = 'application/json';
        }

        const response = await fetch(`${this.baseURL}${endpoint}`, {
            method,
            headers,
            body: body ? JSON.stringify(body) : null
        });

        if (response.status === 401) {
            this.token = null;
            localStorage.removeItem('auth_token');
            throw new Error('Unauthorized');
        }

        return response.json();
    }

    async getMembros() {
        return this.makeRequest('/api/membros');
    }
