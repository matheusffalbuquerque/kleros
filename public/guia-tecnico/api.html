🔌 APIs

Sistema de APIs RESTful, integração com aplicativos móveis e serviços externos.

🎯 Arquitetura das APIs

Como as APIs estão estruturadas no Kleros

Estrutura da API

  • 🏗️ REST: Arquitetura RESTful padrão
  • 🔐 Autenticação: Laravel Sanctum com tokens
  • 📄 Formato: JSON request/response
  • 🎯 Versionamento: /api/v1 prefix
  • 🛡️ Rate Limiting: 60 requests/minuto
  • 📊 Paginação: Link headers + meta

Base URLs

# Produção
https://{congregacao}.kleros.app/api/v1

# Desenvolvimento
http://{congregacao}.kleros.local/api/v1

# Exemplos
https://primeiraigreja.kleros.app/api/v1/membros
http://testecongregacao.kleros.local/api/v1/cultos

Headers padrão:
Accept: application/json
Content-Type: application/json
Authorization: Bearer {token}

Resposta padrão:
{
    "data": [...],
    "message": "Success",
    "status": 200,
    "meta": {
        "total": 150,

        "last_page": 8
    },
    "links": {
        "first": "url",
        "last": "url",

        "next": "url"
    }

    /**
     * Lista paginada de membros
     */
    public function index(Request $request)
    {
        $congregacao = $this->getCongregacao();
        
        $query = $congregacao->membros()
            ->with(['situacao', 'estadoCivil', 'escolaridade']);

        // Filtros
        if ($request->filled('nome')) {
            $query->where('nome', 'like', '%' . $request->nome . '%');
        }

        if ($request->filled('situacao_id')) {
            $query->where('situacao_id', $request->situacao_id);
        }

        if ($request->filled('ativo')) {
            $query->where('ativo', $request->boolean('ativo'));
        }

        // Ordenação
        $sortField = $request->get('sort', 'nome');
        $sortDirection = $request->get('direction', 'asc');
        $query->orderBy($sortField, $sortDirection);

        // Paginação
        $perPage = min($request->get('per_page', 20), 100);
        $membros = $query->paginate($perPage);

        return new MembroCollection($membros);
    }

    /**
     * Exibir membro específico
     */
    public function show(Membro $membro)
    {
        // Verificar se pertence à congregação
        if ($membro->congregacao_id !== $this->getCongregacao()->id) {
            return $this->notFound();
        }

        $membro->load([
            'situacao',
            'estadoCivil', 
            'escolaridade',
            'user',
            'celulas'
        ]);

        return new MembroResource($membro);
    }

    /**
     * Criar novo membro
     */
    public function store(MembroStoreRequest $request)
    {
        $data = $request->validated();
        $data['congregacao_id'] = $this->getCongregacao()->id;

        $membro = Membro::create($data);

        return $this->success(
            new MembroResource($membro),
            'Membro criado com sucesso',
            201
        );
    }

    /**
     * Atualizar membro
     */
    public function update(MembroUpdateRequest $request, Membro $membro)
    {
        // Verificar se pertence à congregação
        if ($membro->congregacao_id !== $this->getCongregacao()->id) {
            return $this->notFound();
        }

        $membro->update($request->validated());

        return $this->success(
            new MembroResource($membro->fresh()),
            'Membro atualizado com sucesso'
        );
    }

    /**
     * Excluir membro
     */
    public function destroy(Membro $membro)
    {
        // Verificar se pertence à congregação
        if ($membro->congregacao_id !== $this->getCongregacao()->id) {
            return $this->notFound();
        }

        // Verificar se pode ser excluído
        if ($membro->user) {
            return $this->error(
                'Não é possível excluir membro com usuário vinculado',
                400
            );
        }

        $membro->delete();

        return $this->success(
            null,
            'Membro excluído com sucesso'
        );
    }

    /**
     * Buscar membros
     */
    public function search(Request $request)
    {
        $request->validate([
            'q' => 'required|string|min:2'
        ]);

        $congregacao = $this->getCongregacao();
        
        $membros = $congregacao->membros()
            ->where('nome', 'like', '%' . $request->q . '%')
            ->where('ativo', true)
            ->limit(10)
            ->get(['id', 'nome', 'email', 'telefone']);

        return $this->success($membros);
    }

    /**
     * Estatísticas dos membros
     */
    public function stats()
    {
        $congregacao = $this->getCongregacao();

        $stats = [
            'total' => $congregacao->membros()->count(),
            'ativos' => $congregacao->membros()->where('ativo', true)->count(),
            'inativos' => $congregacao->membros()->where('ativo', false)->count(),
            'por_situacao' => $congregacao->membros()
                ->join('situacoes_membro', 'membros.situacao_id', '=', 'situacoes_membro.id')
                ->groupBy('situacoes_membro.nome')
                ->selectRaw('situacoes_membro.nome, count(*) as total')
                ->pluck('total', 'nome'),
            'por_sexo' => [
                'masculino' => $congregacao->membros()->where('sexo', 'M')->count(),
                'feminino' => $congregacao->membros()->where('sexo', 'F')->count(),
            ]
        ];

        return $this->success($stats);
    }
}

    private function isValidCpf($cpf): bool
    {
        $cpf = preg_replace('/[^0-9]/', '', $cpf);

        if (strlen($cpf) != 11 || preg_match('/^(\d)\1{10}$/', $cpf)) {
            return false;
        }

        for ($i = 9; $i < 11; $i++) {
            $sum = 0;
            for ($j = 0; $j < $i; $j++) {
                $sum += intval($cpf[$j]) * (($i + 1) - $j);
            }
            $digit = ((10 * $sum) % 11) % 10;
            if (intval($cpf[$i]) !== $digit) {
                return false;
            }
        }

        return true;
    }
}

// Configurar rate limiting no RouteServiceProvider
protected function configureRateLimiting()
{
    RateLimiter::for('api', function (Request $request) {
        return Limit::perMinute(60)->by($request->user()?->id ?: $request->ip());
    });

    RateLimiter::for('login', function (Request $request) {
        return Limit::perMinute(5)->by($request->ip());
    });
}

# Version Middleware
class ApiVersionMiddleware
{
    public function handle($request, Closure $next, $version = 'v1')
    {
        $request->attributes->set('api_version', $version);
        
        $response = $next($request);
        
        $response->headers->set('API-Version', $version);
        $response->headers->set('API-Current-Version', config('api.current_version'));
        
        return $response;
    }
}
