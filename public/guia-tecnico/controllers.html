Gestão completa de membros

class MembroController extends Controller
{
    // Listar membros
    public function index() {
        $membros = Membro::with('celulas')
            ->ativos()
            ->paginate(50);
            
        return view('membros.index', compact('membros'));
    }
    
    // Criar membro
    public function store(Request $request) {
        $validated = $request->validate([
            'primeiro_nome' => 'required|string|max:255',
            'sobrenome' => 'required|string|max:255',
            'email' => 'nullable|email|unique:membros',
            'data_nascimento' => 'nullable|date',
        ]);
        
        $validated['congregacao_id'] = app('congregacao')->id;
        
        Membro::create($validated);
        
        return redirect()->route('membros.index')
            ->with('success', 'Membro cadastrado!');
    }
    
    // Busca dinâmica
    public function search(Request $request) {
        $query = $request->get('q');
        
        $membros = Membro::where('primeiro_nome', 'like', "%{$query}%")
            ->orWhere('sobrenome', 'like', "%{$query}%")
            ->limit(10)
            ->get();
            
        return response()->json($membros);
    }
}

⛪ CongregacaoController

Gestão de congregações (admin)

class CongregacaoController extends Controller
{
    // Cadastro inicial
    public function create() {
        $denominacoes = Denominacao::ativas()->get();
        $paises = Pais::all();
        
        return view('congregacoes.create', compact(
            'denominacoes', 'paises'
        ));
    }
    
    // Criar congregação
    public function store(Request $request) {
        DB::transaction(function () use ($request) {
            // 1. Criar congregação
            $congregacao = Congregacao::create($validated);
            
            // 2. Criar domínio
            Dominio::create([
                'congregacao_id' => $congregacao->id,
                'dominio' => $request->dominio,
                'ativo' => true
            ]);
            
            // 3. Criar usuário gestor
            $user = User::create([
                'name' => $request->gestor_nome,
                'email' => $request->gestor_email,
                'password' => $request->gestor_senha,
                'congregacao_id' => $congregacao->id
            ]);
            
            // 4. Atribuir role
            $user->assignRole('gestor');
            
            // 5. Enviar email de boas-vindas
            Mail::to($user)->send(new CongregacaoGestorBoasVindas($congregacao));
        });
    }
}

class LocalizacaoController extends Controller

        return response()->json($estados);
    }
    
    // API para buscar cidades

class MembroController extends Controller
{
    // Busca AJAX
    public function search(Request $request) {
        $query = $request->get('q', '');
        
        if (strlen($query) < 2) {
            return response()->json([]);
        }
        
        $membros = Membro::where('primeiro_nome', 'like', "%{$query}%")
            ->orWhere('sobrenome', 'like', "%{$query}%")
            ->select('id', 'primeiro_nome', 'sobrenome')
            ->limit(10)
            ->get()
            ->map(function ($membro) {
                return [
                    'id' => $membro->id,
                    'text' => $membro->primeiro_nome . ' ' . $membro->sobrenome
                ];
            });
            
        return response()->json($membros);
    }
}

                "unique:membros,email,NULL,id,congregacao_id,{$congregacaoId}"
            ],

        ];
    }
    
    public function messages()

        return [

        ];
    }

    {
        $this->merge([
            'congregacao_id' => app('congregacao')->id,
        ]);
    }
}

📁 Upload de Arquivos

Gestão de uploads e armazenamento

class ArquivoController extends Controller
{
    public function upload(Request $request)
    {
        $request->validate([
            'arquivo' => 'required|file|max:10240', // 10MB
            'tipo' => 'required|in:documento,imagem,video,audio',
        ]);
        
        $file = $request->file('arquivo');
        $congregacao = app('congregacao');
        
        // Gerar nome único
        $filename = time() . '_' . Str::random(10) . '.' . $file->extension();
        
        // Armazenar no diretório da congregação
        $path = $file->storeAs(
            "congregacoes/{$congregacao->id}/arquivos",
            $filename,
            'public'
        );
        
        // Salvar no banco
        $arquivo = Arquivo::create([
            'congregacao_id' => $congregacao->id,
            'nome' => $file->getClientOriginalName(),
            'caminho' => $path,
            'tipo' => $request->tipo,
        ]);
        
        return response()->json([
            'success' => true,
            'arquivo' => $arquivo,
            'url' => Storage::url($path)
        ]);
    }
    
    public function download(Arquivo $arquivo)
    {
        // Verificar permissão
        if ($arquivo->congregacao_id !== app('congregacao')->id) {
            abort(403);
        }
        
        return Storage::download($arquivo->caminho, $arquivo->nome);
    }
}

class MembroController extends Controller
{
    public function __construct()
    {
        // Middleware para todas as ações
        $this->middleware('auth');
        $this->middleware('dominio');
        
        // Middleware para ações específicas
        $this->middleware('role:gestor')->only([
            'create', 'store', 'edit', 'update', 'destroy'
        ]);
        
        // Middleware exceto certas ações
        $this->middleware('member.activity')->except(['api']);
    }
}
