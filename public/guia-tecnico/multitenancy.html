🏢 Multi-tenancy

Implementação de multi-tenancy baseada em domínios no sistema Kleros, garantindo isolamento de dados entre congregações.

🎯 Conceito e Estratégia

Como funciona o multi-tenancy no Kleros

O que é Multi-tenancy?

Multi-tenancy permite que múltiplas organizações (tenants) compartilhem a mesma aplicação mantendo seus dados isolados e seguros.

Benefícios:
  • 💰 Redução de custos de infraestrutura
  • 🔧 Manutenção centralizada
  • 📈 Escalabilidade otimizada
  • 🔒 Isolamento de dados

⚙️ Implementação Técnica

Como o sistema identifica e isola os tenants

FLUXO DE IDENTIFICAÇÃO DE TENANT:

1. REQUISIÇÃO CHEGA
   ├── User acessa: congregacao1.local/dashboard
   ├── Laravel captura: request()->getHost()
   └── Host = "congregacao1.local"

2. MIDDLEWARE INTERCEPTA
   ├── AcessarCongregacaoPeloDominio.php
   ├── Extrai subdomain: "congregacao1"
   └── Busca na tabela 'dominios'

3. RESOLUÇÃO DO TENANT
   ├── SELECT * FROM dominios WHERE dominio = 'congregacao1'
   ├── JOIN com congregacoes
   └── Verifica se ativa

4. BINDING NO CONTAINER
   ├── app()->singleton('congregacao', $congregacao)
   ├── Disponível em toda aplicação
   └── Controllers acessam via app('congregacao')

5. APLICAÇÃO DO CONTEXTO
   ├── Queries filtradas automaticamente
   ├── Views recebem contexto
   └── URLs geradas com tenant correto

Global Scopes

Filtragem automática via Eloquent

// Scope global para auto-filtrar por tenant
class TenantScope implements Scope
{
    public function apply(Builder $builder, Model $model)
    {
        if (app()->bound('congregacao')) {
            $congregacao = app('congregacao');
            $builder->where($model->getTable() . '.congregacao_id', $congregacao->id);
        }
    }
}

// Aplicar em models que precisam
class Membro extends Model
{
    protected static function booted()
    {
        static::addGlobalScope(new TenantScope);
    }
    
    // Para queries sem filtro (admin)
    public function newQueryWithoutScopes()
    {
        return parent::newQueryWithoutScopes();
    }
}

// Uso automático
$membros = Membro::all(); // Já filtrado por congregacao_id
$todosMembros = Membro::withoutGlobalScopes()->get(); // Sem filtro

{
    public function register()
    {
        // Binding da congregação será feito pelo middleware

    public function boot()
    {
        // Helper global para acessar a congregação

            function congregacao() {
                return app('congregacao');
            }
        }
        
        // Macro para URL com tenant
        \Illuminate\Support\Facades\URL::macro('tenant', function ($route, $parameters = []) {

            return route($route, $parameters, true, $domain);
        });
    }
}

⚙️ Configuração de Domínios

Setup e gestão dos domínios dos tenants

Tabela de Domínios

// Migration: create_dominios_table
Schema::create('dominios', function (Blueprint $table) {
    $table->id();
    $table->foreignId('congregacao_id')
          ->constrained('congregacoes')
          ->onDelete('cascade');
    $table->string('dominio')->unique();
    $table->boolean('ativo')->default(true);
    $table->timestamps();
    
    $table->index(['dominio', 'ativo']);
});

// Model: Dominio
class Dominio extends Model
{
    protected $fillable = [
        'congregacao_id', 'dominio', 'ativo'
    ];
    
    protected $casts = [
        'ativo' => 'boolean'
    ];
    
    public function congregacao()
    {
        return $this->belongsTo(Congregacao::class);
    }
    
    // Scope para domínios ativos
    public function scopeAtivos($query)
    {
        return $query->where('ativo', true);
    }
}

Criação de Domínio

// CongregacaoController - Criar congregação com domínio
public function store(Request $request)
{
    $request->validate([
        'identificacao' => 'required|string|max:255',
        'dominio' => [
            'required',
            'string',
            'max:50',
            'regex:/^[a-z0-9\-]+$/',
            'unique:dominios,dominio'
        ],
        // ... outros campos
    ]);
    
    DB::transaction(function () use ($request) {
        // 1. Criar congregação
        $congregacao = Congregacao::create([
            'identificacao' => $request->identificacao,
            'denominacao_id' => $request->denominacao_id,
            'cidade_id' => $request->cidade_id,
            'ativa' => true
        ]);
        
        // 2. Criar domínio
        Dominio::create([
            'congregacao_id' => $congregacao->id,
            'dominio' => $request->dominio,
            'ativo' => true
        ]);
        
        // 3. Criar usuário gestor
        $user = User::create([
            'name' => $request->gestor_nome,
            'email' => $request->gestor_email,
            'password' => Hash::make($request->gestor_senha),
            'congregacao_id' => $congregacao->id
        ]);
        
        $user->assignRole('gestor');
    });
    
    return redirect()->route('congregacoes.sucesso')
        ->with('success', 'Congregação criada com sucesso!');
}

🔒 Segurança Multi-tenant

Garantias de isolamento e proteção

Validações de Segurança

// Model Binding com validação de tenant
Route::bind('membro', function ($id) {
    $congregacao = app('congregacao');
    
    return Membro::where('id', $id)
        ->where('congregacao_id', $congregacao->id)
        ->firstOrFail();
});

// Policy baseada em tenant
class MembroPolicy
{
    public function view(User $user, Membro $membro)
    {
        // Verificar se membro pertence à congregação do usuário
        return $user->congregacao_id === $membro->congregacao_id;
    }
    
    public function update(User $user, Membro $membro)
    {
        return $user->hasRole('gestor') 
            && $user->congregacao_id === $membro->congregacao_id;
    }
}

// Middleware de verificação de tenant
class VerifyTenantAccess
{
    public function handle($request, Closure $next)
    {
        $user = auth()->user();
        $congregacao = app('congregacao');
        
        // Verificar se usuário pertence ao tenant
        if ($user->congregacao_id !== $congregacao->id) {
            abort(403, 'Acesso negado a esta congregação.');
        }
        
        return $next($request);
    }
}

// Middleware para log de atividades
class LogMemberActivity
{
    public function handle($request, Closure $next)
    {
        $response = $next($request);
        
        if (auth()->check()) {
            MemberActivityLog::create([
                'user_id' => auth()->id(),
                'congregacao_id' => app('congregacao')->id,
                'action' => $request->method(),
                'route' => $request->route()->getName(),
                'ip_address' => $request->ip(),
                'user_agent' => $request->userAgent(),
                'created_at' => now()
            ]);
        }
        
        return $response;
    }
}

⚡ Performance em Multi-tenancy

Otimizações específicas para ambiente multi-tenant

Caching por Tenant

// Helper para cache específico do tenant
function tenantCache($key, $callback = null, $ttl = 3600)
{
    $congregacao = app('congregacao');
    $tenantKey = "tenant_{$congregacao->id}_{$key}";
    
    if ($callback) {
        return Cache::remember($tenantKey, $ttl, $callback);
    }
    
    return Cache::get($tenantKey);
}

// Uso em controllers
class DashboardController extends Controller
{
    public function index()
    {
        $stats = tenantCache('dashboard_stats', function () {
            return [
                'membros' => Membro::count(),
                'celulas' => Celula::count(),
                'cultos_mes' => Culto::whereMonth('data_hora', now()->month)->count(),
            ];
        }, 1800); // 30 minutos
        
        return view('dashboard', compact('stats'));
    }
}

// Invalidação de cache
class MembroController extends Controller

Índices de Banco

// Índices otimizados para queries multi-tenant
Schema::create('membros', function (Blueprint $table) {
    $table->id();
    $table->foreignId('congregacao_id')->constrained();
    $table->string('primeiro_nome');
    $table->string('sobrenome');
    $table->boolean('ativo')->default(true);
    $table->timestamps();
    
    // Índices compostos para performance
    $table->index(['congregacao_id', 'ativo']);
    $table->index(['congregacao_id', 'created_at']);
    $table->index(['congregacao_id', 'primeiro_nome', 'sobrenome']);
});

// Query otimizada
class MembroController extends Controller
{
    public function search(Request $request)
    {
        $congregacao = app('congregacao');
        $termo = $request->get('q');
        
        // Query otimizada com índices
        $membros = Membro::select('id', 'primeiro_nome', 'sobrenome')
            ->where('congregacao_id', $congregacao->id) // Primeiro filtro
            ->where('ativo', true)                      // Segundo filtro
            ->where(function ($query) use ($termo) {    // Busca textual
                $query->where('primeiro_nome', 'like', "%{$termo}%")
                      ->orWhere('sobrenome', 'like', "%{$termo}%");
            })
            ->limit(20)
            ->get();
            
        return response()->json($membros);
    }
}
